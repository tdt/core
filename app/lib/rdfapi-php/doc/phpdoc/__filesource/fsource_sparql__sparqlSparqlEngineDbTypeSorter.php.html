  55.     /**
  56.     *   Returns an array of sql statements that need to be executed
  57.     *   and deliver the full result set when combined.
  58.     *
  59.     *   Execute setData() before this.
  60.     *
  61.     *   @internal It is not possible (ok, it is - but it would really complicate
  62.     *    the queries and the code here) to use UNION to combine the sqls to
  63.     *    a single one - UNION merges the results and does not keep the order
  64.     *    of the rows. Citing the mysql manual:
  65.     *        Use of ORDER BY for individual SELECT statements implies nothing
  66.     *        about the order in which the rows appear in the final result because
  67.     *        UNION by default produces an unordered set of rows. If ORDER BY
  68.     *        appears with LIMIT, it is used to determine the subset of the
  69.     *        selected rows to retrieve for the SELECT, but does not necessarily
  70.     *        affect the order of those rows in the final UNION result.
  71.     *        If ORDER BY appears without LIMIT in a SELECT, it is
  72.     *        optimized away because it will have no effect anyway.
  73.     *
  74.     *   @param string $strSelect    SELECT clause
  75.     *   @param string $strFrom      FROM clause
  76.     *   @param string $strWhere     WHERE clause
  77.     *   @return array   Array of arrays. Imploding an array will give
  78.     *                     a complete sql statement. The array will have the keys
  79.     *                     select/from/where/order.
  80.     */
  81.     public function getOrderifiedSqls($arSqls)
  82.     {
  83.         if (count($arSqls) == 0) {
  84.             return $arSqls;
  85.         }
  86.  
  87.         $strSelect = $arSqls[0]['select'];
  88.         $strFrom   = $arSqls[0]['from'];
  89.         $strWhere  = $arSqls[0]['where'];
  90.  
  91.         $strResultForm = $this->query->getResultForm();
  92.         if ($strResultForm == 'ask' || $strResultForm == 'count') {
  93.             return array(
  94.                 $arSqls
  95.             );
  96.         }
  97.  
  98.         $arSpecial = $this->getSpecialOrderVariables();
  99.         if (count($arSpecial) == 0) {
  100.             $strOrder = $this->getSqlOrderBy();
  101.             foreach ($arSqls as &$arSql) {
  102.                 $arSql['order'] = $strOrder;
  103.             }
  104.             return array(
  105.                 $arSqls
  106.             );
  107.         }
  108.  
  109.         $arTypeSets = $this->orderTypeSets(
  110.             $this->getTypeSets($arSpecial, $strFrom, $strWhere)
  111.         );
  112.  
  113.         $arNewSqls = array();
  114.         foreach ($arTypeSets as $arTypeSet) {
  115.             $arSqlParts = array();
  116.             foreach ($arSqls as $arSql) {
  117.                 $arSql['where'] .= $this->getTypesetWhereClause($arTypeSet);
  118.                 $arSql['order']  = $this->getSqlOrderBy($arTypeSet);
  119.                 $arSqlParts[] = $arSql;
  120.             }
  121.             $arNewSqls[] = $arSqlParts;
  122.         }
  123.  
  124.         return $arNewSqls;
  125.     }//public function getOrderifiedSqls($arUsedVarAssignments, $strSelect, $strFrom, $strWhere)

  189.     /**
  190.     *   Determines the type sets in the query results.
  191.     *   A type set is a distinct set of variables and their types,
  192.     *   e.g. the variable and its type (r/b/l) and its datatype
  193.     *   if its an object.
  194.     *
  195.     *   @param array $arSpecialVars     Special variables as returned by
  196.     *                                    getSpecialOrderVariables()
  197.     *   @param string $strFrom          FROM part of the sql query
  198.     *   @param string $strWhere         WHERE part of the sql query
  199.     *
  200.     *   @return array   Key is the sparql variable name, value is an
  201.     *                    array. This one has one key 'type' with a
  202.     *                    value of b/l/r. It might have another key
  203.     *                    'datatype' with the resource's datatype.
  204.     */
  205.     protected function getTypeSets($arSpecialVars, $strFrom, $strWhere)
  206.     {
  207.         $arSel = array();
  208.         foreach ($arSpecialVars as $strSparqlVar) {
  209.             if ($this->arVarAssignments[$strSparqlVar][1] == 'o') {
  210.                 $arSel[] = $this->arVarAssignments[$strSparqlVar][0] . '.l_datatype'
  211.                     . ' as "' . $strSparqlVar . '-datatype"';
  212.             }
  213.             $strTypeCol = $this->arVarAssignments[$strSparqlVar][1] == 'o'
  214.                 ? 'object_is' : 'subject_is';
  215.             $this->arVarAssignments[$strSparqlVar][2] = $strTypeCol;
  216.             $arSel[] = $this->arVarAssignments[$strSparqlVar][0] . '.' . $strTypeCol
  217.                 . ' as "' . $strSparqlVar . '-type"';
  218.         }
  219.  
  220.         $oldmode = $this->dbConn->SetFetchMode(ADODB_FETCH_ASSOC);
  221.         $arResult = $this->dbConn->execute(
  222.             'SELECT DISTINCT ' . implode(', ', $arSel) . ' ' . $strFrom . $strWhere
  223.         );
  224.         $this->dbConn->SetFetchMode($oldmode);
  225.  
  226.         $arTypes = array();
  227.         foreach ($arResult as $arRow) {
  228.             $nLine = count($arTypes);
  229.             foreach ($arRow as $key => $value) {
  230.                 list($strSparqlVar, $strType) = explode('-', $key);
  231.                 $arTypes[$nLine][$strSparqlVar][$strType] = $value;
  232.             }
  233.         }
  234.  
  235.         return $arTypes;
  236.     }//protected function getTypeSets($arSpecialVars, $strFrom, $strWhere)

  240.     /**
  241.     *   Takes an array of type sets (as returned by getTypeSets()) and
  242.     *   sorts the variables according to the SPARQL specs.
  243.     *
  244.     *   @param array $arTypes Array of type sets
  245.     *   @return array Ordered array of type sets
  246.     */
  247.     protected function orderTypeSets($arTypes)
  248.     {
  249.         $arSM = $this->query->getSolutionModifier();
  250.         if ($arSM['order by'] !== null) {
  251.             $this->arDirection = array();
  252.             foreach ($arSM['order by'] as $arVar) {
  253.                 $this->arDirection[$arVar['val']] =
  254.                     (strtoupper($arVar['type']) == 'ASC' ? 1 : -1);

  258.         usort($arTypes, array($this, 'compareTypeSet'));

  265.     /**
  266.     *   Compares two type sets. Works like a normal comparision
  267.     *   method that returns -1/0/1 for use in usort()
  268.     *
  269.     *   @param array $arTs1     First typeset
  270.     *   @param array $arTs2     Second typeset
  271.     *   @return int Comparison value
  272.     */
  273.     public function compareTypeSet($arTs1, $arTs2)
  274.     {
  275.         foreach ($arTs1 as $strSparqlVar => $arType1) {
  276.             //compare types
  277.             $n = self::$arTypeNumbers[$arType1['type']]
  278.                 - self::$arTypeNumbers[$arTs2[$strSparqlVar]['type']];
  279.             if ($n != 0) {
  280.                 //and take ASC/DESC into account
  281.                 return $n * $this->arDirection[$strSparqlVar];
  282.             }
  283.  
  284.             //compare the datatypes
  285.             $n = self::compareXsdType(
  286.                 $arType1['datatype'],
  287.                 $arTs2[$strSparqlVar]['datatype']
  288.             );
  289.             //if they are not equal, return the value
  290.             if ($n != 0) {
  291.                 //and take ASC/DESC into account
  292.                 return $n * $this->arDirection[$strSparqlVar];
  293.             }
  294.         }
  295.  
  296.         //they are equal
  297.         return 0;
  298.     }//public function compareTypeSet($arTs1, $arTs2)

  336.     /**
  337.     *   Returns the ORDER BY sql query string
  338.     *   if neccessary for the query. The returned string
  339.     *   already has ORDER BY prefixed.
  340.     *
  341.     *   @internal Also takes care of casting the data if the type
  342.     *    is listed in $arCastTypes. See getCastMethod() for more info.
  343.     *
  344.     *   @param arrray $arTypeSet Single typeset
  345.     *   @return string      ORDER BY ... string or empty string
  346.     */
  347.     function getSqlOrderBy($arTypeSet = array())
  348.     {
  349.         $arSM = $this->query->getSolutionModifier();
  350.         if ($arSM['order by'] === null) {
  351.             return '';
  352.         }
  353.  
  354.         $sqlOrder = array();
  355.         foreach ($arSM['order by'] as $arVar) {
  356.             $strSparqlVar = $arVar['val'];
  357.             if (isset($this->arUsedVarAssignments[$strSparqlVar])) {
  358.                 if (!isset($arTypeSet[$strSparqlVar]['datatype'])
  359.                   || $arTypeSet[$strSparqlVar]['datatype'] == '') {
  360.                     $sqlOrder[] = $this->arUsedVarAssignments[$strSparqlVar] . ' ' . strtoupper($arVar['type']);
  361.                 } else {
  362.                     $sqlOrder[] = self::getCastMethod(
  363.                         $arTypeSet[$strSparqlVar]['datatype'],
  364.                         $this->arUsedVarAssignments[$strSparqlVar]
  365.                     ) . ' ' . strtoupper($arVar['type']);
  366.                 }
  367.             }
  368.         }
  369.  
  370.         return ' ORDER BY ' . implode(', ', $sqlOrder);
  371.     }//function getSqlOrderBy($arTypeSet = array())

  375.     /**

  396.     /**
  397.     *   Creates and returns the SQL WHERE clauses needed to get only

  403.     protected function getTypesetWhereClause($arTypeSet)

  407.             //check type

  409.                 . '.' . $this->arVarAssignments[$strSparqlVar][2]

  411.             //check datatype

  413.                 $strWhereTypes .= ' AND ' . $this->arVarAssignments[$strSparqlVar][0]
  414.                     . '.l_datatype '
  415.                     . $this->getStringNullComparison($arType['datatype']);
  416.             }
  417.         }
  418.         return $strWhereTypes;
  419.     }//protected function getTypesetWhereClause($arTypeSet)
